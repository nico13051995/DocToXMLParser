123 void FindReplaceForm::find(bool next) {
124  if (!textEdit)
125  return; // TODO: show some warning?
126 
127  // backward search
128  bool back = !next;
129 
130  const QString &toSearch = ui->textToFind->text();
131 
132  bool result = false;
133 
134  QTextDocument::FindFlags flags;
135 
136  if (back)
137  flags |= QTextDocument::FindBackward;
138  if (ui->caseCheckBox->isChecked())
139  flags |= QTextDocument::FindCaseSensitively;
140  if (ui->wholeCheckBox->isChecked())
141  flags |= QTextDocument::FindWholeWords;
142 
143  if (ui->regexCheckBox->isChecked()) {
144  QRegExp reg(toSearch,
145  (ui->caseCheckBox->isChecked() ? Qt::CaseSensitive : Qt::CaseInsensitive));
146 
147  qDebug() << "searching for regexp: " << reg.pattern();
148 
149  textCursor = textEdit->document()->find(reg, textCursor, flags);
150  textEdit->setTextCursor(textCursor);
151  result = (!textCursor.isNull());
152  } else {
153  qDebug() << "searching for: " << toSearch;
154 
155  result = textEdit->find(toSearch, flags);
156  }
157 
158  if (result) {
159  showError("");
160  } else {
161  showError(tr("no match found"));
162  // move to the beginning of the document for the next find
163  textCursor.setPosition(0);
164  textEdit->setTextCursor(textCursor);
165  }
166 }
